<pre>
struct Node {
  int val;
  Node *left;
  Node *right;
  Node *next;
}</pre>

        traverse(root->left, root->right);

func connect(root *Node) *Node {
    if root == nil {
        return nil
    }
    // 二叉树层序遍历框架
    q := make([]*Node, 0)
    q = append(q, root)
    for len(q) > 0 {
        sz := len(q)
        // 遍历一层
        var pre *Node
        for i := 0; i < sz; i++ {
            cur := q[0]
            q = q[1:]
            // 链接当前层所有节点的 next 指针
            if pre != nil {
                pre.Next = cur
            }
            pre = cur
            // 将下一层节点装入队列
            if cur.Left != nil {
                q = append(q, cur.Left)
            }
            if cur.Right != nil {
                q = append(q, cur.Right)
            }
        }
    }
    return root
}

    q.push(root);
